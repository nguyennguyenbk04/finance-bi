https://www.kaggle.com/datasets/computingvictor/transactions-fraud-datasets


# 1. Start all services
cd /home/bnguyen/Desktop/finance_analytics/infra
sudo docker compose up -d

# 2. Verify services are running
docker compose ps

# 3. Register connector
curl -i -X POST -H "Accept:application/json" -H "Content-Type:application/json"\
 localhost:8083/connectors/ \
  -d @/home/bnguyen/Desktop/finance_analytics/infra/config/register-mysql.json

# 4. Delete connector
curl -i -X DELETE localhost:8083/connectors/finance-mysql-connector

# 5. Check status
curl -s localhost:8083/connectors/finance-mysql-connector/status | jq

# 6. Check Kafka topics
sudo docker exec kafka /kafka/bin/kafka-topics.sh --bootstrap-server kafka:9092 --list

# 7. Access MySQL
# MySQL Workbench: localhost:30306, user: root, password: root123
# Or via Docker: sudo docker exec -it mysql mysql -u root -p
